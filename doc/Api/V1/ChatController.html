def create
  chat = Chat.new(chat_params)
  chat.save!
  render json: chat, status: :created
rescue StandardError => e
  render json: e, status: :bad_request
end

def delete

  render json: chat, status: :ok
rescue StandardError => e
  render json: e, status: :bad_request
end
